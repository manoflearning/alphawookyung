
#include <bits/stdc++.h>

const double EPS = 1e-9;
const int INF = 1e9 + 7;
const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int tc; cin >> tc;
	for (int tt = 1; tt <= tc; tt++) {
		ll n, x, y;
        cin >> n >> x >> y;

        __int128_t ans = 0;
        while (n > 0) {
            if (n & 1) {
                ans += x, n--;
                continue;
            }

            __int128_t nxt = n / 2;
            ans += min<__int128_t>((n - nxt) * x, y);
            n = nxt;
        }

        cout << (ll)ans << '\n';
	}

	return 0;
}
