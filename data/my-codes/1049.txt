const double EPS = 1e-9;

ll gcd(ll a, ll b) {
	if (b == 0) return a;
	else return gcd(b, a % b);
}

ll lcm(ll a, ll b) {
	return a * b / gcd(a, b);
}

ll a, as, ae;
ll b, bs, be;
ll c, cs, ce;

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> a >> as >> ae;
    cin >> b >> bs >> be;
    cin >> c >> cs >> ce;

    ll g = lcm(a, lcm(b, c));

    ll ans = -1;
    for (ll x = 0; x < g; ) {
        ll ma = x % a, mb = x % b, mc = x % c;
        
        ll nxt = 1;
        int bit = 1;

        if (ma < as) nxt = max(nxt, as - ma), bit = 0;
        if (ae < ma) nxt = max(nxt, a - ma + as), bit = 0;
        if (mb < bs) nxt = max(nxt, bs - mb), bit = 0;
        if (be < mb) nxt = max(nxt, b - mb + bs), bit = 0;
        if (mc < cs) nxt = max(nxt, cs - mc), bit = 0;
        if (ce < mc) nxt = max(nxt, c - mc + cs), bit = 0;

        if (bit) {
            ans = x;
            break;
        }

        x += nxt;
    }

    cout << ans;

	return 0;
}
