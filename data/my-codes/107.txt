ll n, k, a, b;

ll binom(int n, int r) {
    if (r < 0 || n < r) return 0;

	r = min(r, n - r);
	ll ret = 1;
	for (ll i = 1; i <= r; i++) {
		ret *= n + 1 - i;
		ret /= i;
	}
	return ret;
}

ll cal(ll x) {
    ll ret = 0;

    ll cnt = 0;
    for (int i = 60; i >= 0; i--) {
        ll bit1 = n & (1ll << i);
        ll bit2 = x & (1ll << i);
        
        if (bit2) {
            if (bit1 == 0) ret += binom(i, k - cnt);
            else ret += binom(i, k - cnt - 1);
        }

        if (bit1 ^ bit2) cnt++;
    }
    if (cnt == k) ret++;

    return ret;
}
