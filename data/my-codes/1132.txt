#include <bits/stdc++.h>
using namespace std;

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

    string s; cin >> s;

    int ans = 0;
    int cnt = 0;
    for (auto& i : s) {
        if (i == '(') cnt++;
        if (i == ')') {
            if (cnt) cnt--;
            else ans++;
        }
    }
    ans += cnt;

    cout << ans;
}
