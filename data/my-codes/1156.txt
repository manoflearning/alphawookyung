void hanoi(int count, char from, char tmp, char to) {
	if (count == 1) {
		printf("%c %c\n", from, to);
		return;
	}
	hanoi(count - 1, from, to, tmp);
	printf("%c %c\n", from, to);
	hanoi(count - 1, tmp, from, to);
}
int hanoi_count(int x) {
	int count = 0;
	for (int i = 0; i < x; i++) {
		count += pow(2, i);
	}

	return count;
}
