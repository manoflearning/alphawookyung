#include <bits/stdc++.h>
using namespace std;
#define ll long long

struct Node { ll h, w, p; };

int n;
ll h;
vector<Node> a;
ll dp[1 << 20];

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    cin >> n >> h;
    a.resize(n);
    for (auto& i : a)
        cin >> i.h >> i.w >> i.p;

    memset(dp, -1, sizeof(dp));
    dp[0] = 1e9 + 7;
    ll ans = -1;
    for (int i = 1; i < (1 << n); i++) {
        ll sumh = 0;
        for (int j = 0; j < n; j++) {
            if (!(i & (1 << j))) continue;
            if (dp[i - (1 << j)] >= a[j].w) {
                dp[i] = max(dp[i], min(dp[i - (1 << j)] - a[j].w, a[j].p));
            }
            sumh += a[j].h;
        }

        if (sumh >= h) ans = max(ans, dp[i]);
    }
    if (ans == -1) cout << "Mark is too tall";
    else cout << ans;
}
