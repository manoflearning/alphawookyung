int flag;  
struct Seg {  
	vector<ll> t;

	void build(int n) {
		for (flag = 1; flag < n; flag <<= 1);
		t.resize(2 * flag);
	}
	void modify(int p) {  
		for (t[p += flag - 1] = 1; p > 1; p >>= 1) t[p >> 1] = t[p] + t[p ^ 1];
	}
	ll query(int l, int r, int n = 1, int nl = 1, int nr = flag) {  
		if (r < nl || nr < l) return 0;
		if (l <= nl && nr <= r) return t[n];

		int mid = (nl + nr) / 2;
		return query(l, r, n << 1, nl, mid) + query(l, r, n << 1 | 1, mid + 1, nr);
	}
}seg;

int n;
vt<pair<string, int>> a, b;
pii arr[101010];

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n;

    FOR(n) {
        string s, rs; cin >> s;
        a.push_back({ s, i });
        FOR(i, sz(s) - 1, -1, -1) rs.push_back(s[i]);
        b.push_back({ rs, i });
    }

    sort(all(a));
    sort(all(b));
    FOR(n) {
        arr[a[i].sc].fr = i + 1;
        arr[b[i].sc].sc = i + 1;
    }

    sort(arr, arr + n);

    ll ans = 0;
    seg.build(n);
    FOR(n) {
        ans += seg.query(arr[i].sc + 1, n);
        seg.modify(arr[i].sc);
    }

    cout << ans;

	return 0;
}
