ll x;

void f(ll x) {
    if (x == 0) return;

    ll lb = sqrt(x);

    for (int i = 0; i < lb; i++) cout << 'A';
    for (int i = 1; i < 25; i++) cout << (char)('A' + i);
    for (int i = 0; i < lb; i++) cout << 'Z';

    f(x - lb * lb);
}
