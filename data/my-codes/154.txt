int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	string s; cin >> s;

    ll ans = 0;
    for (int i = 0; i < sz(s); i++) {
        if (s[i] != 'A') continue;
        
        int imp = 0;
        ll cntA = 1, cntZ = 0;
        char lst = s[i];
        for (int j = i + 1; j < sz(s); j++) {
            if (!(s[j] == lst || s[j] == lst + 1)) break;

            i = j;

            lst = s[j];
            if (lst == 'A') cntA++;
            if (lst == 'Z') cntZ++;
        }
        
        ans += cntA * cntZ;
    }

    cout << ans;

	return 0;
}
