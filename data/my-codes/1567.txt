#include <bits/stdc++.h>
using namespace std;

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    int n, t, p;
    cin >> n >> t >> p;

    vector<int> a(n);
    for (auto& i : a) cin >> i;

    int ans = 0, tleft = t;
    priority_queue<int> pq;
    int res = 0;
    for (int i = 0; i < n; i++) {
        pq.push(a[i]);
        tleft -= a[i];
        res++;
        while (tleft < 0 && !pq.empty()) {
            tleft += pq.top(); pq.pop();
            res--;
        }
        if (tleft < 0) break;
        ans = max(ans, res);
        tleft -= p;
    }

    cout << ans;
}
