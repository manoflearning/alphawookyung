#include <bits/stdc++.h>
using namespace std;
#define ll long long

int n;
vector<ll> a;

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    cin >> n;
    a.resize(n);
    for (auto& i : a) cin >> i;

    sort(a.begin(), a.end());

    ll ans = 0;
    for (int i = 0; i < n - 1; i++) {
        ans = max(ans, min(a[n - 1] - a[i], a[i]));
    }

    cout << ans;
}
