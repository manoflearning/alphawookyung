#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pll pair<ll, ll>
#define fr first
#define sc second

pair<pll, ll> egcd(ll a, ll b) { 
    ll s = 0, olds = 1;
    ll t = 1, oldt = 0;
    ll r = b, oldr = a;

    while (r != 0) {
        ll q = oldr / r;
        ll tmp = oldr - q * r;
        oldr = r, r = tmp;
        tmp = olds - q * s;
        olds = s, s = tmp;
        tmp = oldt - q * t;
        oldt = t, t = tmp;
    }

    return { { olds, oldt }, oldr };
}

ll modInv(ll a, ll mod) {
    pair<pll, ll> res = egcd(a, mod);
    if (res.sc == 1) return (res.fr.fr + mod) % mod;
    else return -1;
}

int main() {
    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    ll N, A;
    cin >> N >> A;

    cout << N - A << ' ' << modInv(A, N);

    return 0;
}
