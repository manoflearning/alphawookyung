const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

ll dp[1010101], psum[1010101];

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	dp[0] = 1;
    dp[1] = 2;
	dp[2] = 7;
	psum[0] = 1;
	psum[1] = 3;
	psum[2] = 10;
    for (int i = 3; i < 1010101; i++) {
        dp[i] = 2 * psum[i - 3] + 3 * dp[i - 2] + 2 * dp[i - 1];
        dp[i] %= MOD;

		psum[i] = dp[i] + psum[i - 1];
		psum[i] %= MOD;
    }

    int n; cin >> n;
    cout << dp[n];

	return 0;
}
