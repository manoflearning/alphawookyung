struct frac {
    ll a, b;

    ll gcd(ll x, ll y) {
	    if (y == 0) return x;
	    else return gcd(y, x % y);
    }
    void trim() {
        ll g = gcd(a, b);
        a /= g, b /= g;
    }
};

int n;
vt<frac> arr;
vt<ll> arr2;

void input() {
    cin >> n;
    arr.resize(n);
    EACH(i, arr) {
        cin >> i.a >> i.b;
        i.trim();
    }
}

ll gcd(ll a, ll b) {
	if (b == 0) return a;
	else return gcd(b, a % b);
}

ll lcm(ll a, ll b) {
	return a * b / gcd(a, b);
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	input();

    ll x = 1;
    EACH(i, arr) x = lcm(x, i.b);

    EACH(i, arr) {
        arr2.push_back(i.a * x / i.b);
    }

    ll y = arr2[0];
    EACH(i, arr2) y = gcd(y, i);

    ll g = gcd(x, y);

    cout << y / g << ' ' << x / g;

	return 0;
}
