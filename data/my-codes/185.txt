int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int n; cin >> n;
    
    int ans = 0;
    vector<int> vi(202020);
    for (int i = 0; i < n; i++) {
        int x, y;
        cin >> x >> y;
        if (vi[x] && y == 1) ans++;
        if (!vi[x] && y == 0) ans++;

        vi[x] = y;
    }

    for (int i = 1; i < 202020; i++) {
        if (vi[i]) ans++;
    }

    cout << ans;

	return 0;
}
