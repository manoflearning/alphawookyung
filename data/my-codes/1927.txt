	int t;
	scanf("%d", &t);

	for (int test = 0; test < t; test++) {
		int a, b, c, d, e;
		scanf("%d %d %d %d %d", &a, &b, &c, &d, &e);

		double sum = 0;
		sum += 350.34 * a;

}
