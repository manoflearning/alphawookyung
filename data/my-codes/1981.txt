int n, dep[101010], en;
vector<int> t[101010];

void dfs(int v) {
    if (t[v][0] != -1) {
        dep[t[v][0]] = dep[v] + 1;
        dfs(t[v][0]);
    }
    en = v;
    if (t[v][1] != -1) {
        dep[t[v][1]] = dep[v] + 1;
        dfs(t[v][1]);
    }
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n;
    for (int i = 0; i < n; i++) {
        int a, b, c;
        cin >> a >> b >> c;
        t[a].push_back(b);
        t[a].push_back(c);
    }

    dfs(1);
    
    cout << 2 * (n - 1) - dep[en];

	return 0;
}
