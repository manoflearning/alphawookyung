struct lf { 
    ll p, q;
    double s;
    lf() : lf(1, 0) {}
    lf(ll sp, ll sq) : p(sp), q(sq), s(0) {}
};

double cross(const lf& u, const lf& v) {
    return (double)(v.q - u.q) / (u.p - v.p);
}

int n;
ll a, b, c;
ll x[1010101], p[1010101], dp[1010101];
lf ch[1010101];

ll cal(ll x) {
    return a * x * x + b * x + c;
}

void convexHullTrick() {
    int top = 1;
    
    for (int i = 1; i <= n; i++) {
        lf g(-2 * a * p[i - 1] + b, a * p[i - 1] * p[i - 1] - b * p[i - 1] + dp[i - 1]);
        while (top > 1) {
            g.s = cross(ch[top - 1], g);
            if (ch[top - 1].s < g.s) break;
            --top;
        }
        ch[top++] = g;

        int l = 1, r = top - 1;
        while (l < r) {
            int mid = (l + r + 1) >> 1;
            if (p[i] < ch[mid].s) r = mid - 1;
            else l = mid;
        }

        int fpos = l;
        dp[i] = ch[fpos].p * p[i] + ch[fpos].q + a * p[i] * p[i] + c;
    }

}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n >> a >> b >> c;
    for (int i = 1; i <= n; i++) {
        cin >> x[i];
        p[i] = x[i] + p[i - 1];
    }

    convexHullTrick();

    cout << dp[n];

	return 0;
}
