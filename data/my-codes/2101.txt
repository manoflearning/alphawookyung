#include <bits/stdc++.h>
using namespace std;
#define pii pair<int, int>
#define fr first
#define sc second

const int MAXN = 101010;

int n, m, a[MAXN];
int r, vi[MAXN];
vector<pii> adj[MAXN];
priority_queue<pii, vector<pii>, greater<pii>> pq;

void input() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> a[i];
    cin >> r;
    for (int i = 0; i < r; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({ w, v });
    }
}

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    input();

    for (int i = 1; i <= n; i++) {
        pq.push({ a[i], i });
    }

    int ans = 0, cnt = 0;
    while (cnt < m) {
        pii res = pq.top();
        pq.pop();
        if (vi[res.sc]) continue;
        ans = max(ans, res.fr);
        cnt++;
        vi[res.sc] = 1;
        for (auto& i : adj[res.sc]) {
            if (vi[i.sc]) continue;
            a[i.sc] -= i.fr;
            pq.push({ a[i.sc], i.sc });
        }
    }

    cout << ans;
}
