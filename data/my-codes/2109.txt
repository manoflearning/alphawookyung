#define EXPAND( x ) x 
#define F_OR(i, a, b, s) for (int i = (a); (s) > 0 ? i < (b) : i > (b); i += (s))
#define F_OR1(e) F_OR(i, 0, e, 1)
#define F_OR2(i, e) F_OR(i, 0, e, 1)

#define F_OR4(i, b, e, s) F_OR(i, b, e, s)

const double EPS = 1e-9;
const int INF = 1e9 + 7;
const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

struct Seg {
	int flag;  
	vector<ll> t;

	void build(int N, vt<int>& a) {
		for (flag = 1; flag < N; flag <<= 1);
		t.resize(2 * flag);

		for (int i = flag; i < flag + N; i++) t[i] = a[i - flag + 1];
		for (int i = flag - 1; i >= 1; i--) t[i] = max(t[i << 1], t[i << 1 | 1]);
	}
	void modify(int p, ll value) {  
		for (t[p += flag - 1] = value; p > 1; p >>= 1) t[p >> 1] = max(t[p], t[p ^ 1]);
	}
	ll query(int l, int r) {
		return query(l, r, 1, 1, flag);
	}
	ll query(int l, int r, int n, int nl, int nr) {  
		if (r < nl || nr < l) return 0;
		if (l <= nl && nr <= r) return t[n];

		int mid = (nl + nr) / 2;
		return max(query(l, r, n << 1, nl, mid), query(l, r, n << 1 | 1, mid + 1, nr));
	}
}seg[505][2];

int n, m;
vt<vt<int>> a;
map<int, int> mp;

void input() {
    cin >> n >> m;
    a.resize(n + 1, vt<int>(m + 1));
    FOR(i, 1, n + 1) {
        FOR(j, 1, m + 1) {
            cin >> a[i][j];
        }
    }
}

void makingSeg() {
    FOR(i, 1, n + 1) {
        seg[i][0].build(m, a[i]);
    }
    FOR(i, 1, m + 1) {
        vt<int> b;
        FOR(j, n + 1) b.push_back(a[j][i]);
        seg[i][1].build(n, b);
    }
}

void push(int x) {
    auto it = mp.find(x);
    if (it != mp.end()) it->sc++;
    else mp.insert({ x, 1 });
}

void del(int x) {
    auto it = mp.find(x);
    if (it->sc == 1) mp.erase(x);
    else it->sc--;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	input();

    makingSeg();

    int q; cin >> q;

    while (q--) {
        int x, y, r, c, k, d;
        cin >> x >> y >> r >> c >> k >> d;

        mp.clear();

        int res = 0;
        if (d == 1) {
            FOR(i, x, x + r) {
                push(seg[i][0].query(y, y + c - 1));
            }

            res = (--mp.end())->fr;

            FOR(i, k - 1) {
                push(seg[x + r + i][0].query(y, y + c - 1));
                del(seg[x + i][0].query(y, y + c - 1));
                res ^= (--mp.end())->fr;
            }
        }
        if (d == 2) {
            FOR(i, x, x + r) {
                push(seg[i][0].query(y, y + c - 1));
            }

            res = (--mp.end())->fr;

            FOR(i, k - 1) {
                push(seg[x - i - 1][0].query(y, y + c - 1));
                del(seg[x + r - i - 1][0].query(y, y + c - 1));
                res ^= (--mp.end())->fr;
            }
        }
        if (d == 3) {
            FOR(i, y, y + c) {
                push(seg[i][1].query(x, x + r - 1));
            }

            res = (--mp.end())->fr;

            FOR(i, k - 1) {
                push(seg[y + c + i][1].query(x, x + r - 1));
                del(seg[y + i][1].query(x, x + r - 1));
                res ^= (--mp.end())->fr;
            }
        }
        if (d == 4) {
            FOR(i, y, y + c) {
                push(seg[i][1].query(x, x + r - 1));
            }

            res = (--mp.end())->fr;

            FOR(i, k - 1) {
                push(seg[y - i - 1][1].query(x, x + r - 1));
                del(seg[y + c - i - 1][1].query(x, x + r - 1));
                res ^= (--mp.end())->fr;
            }
        }

        cout << res << '\n';
    }

	return 0;
}
