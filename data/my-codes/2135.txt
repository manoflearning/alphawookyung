
#include <bits/stdc++.h>
using namespace std;
#define ll long long

int n;
vector<int> a, b;

void input() {
    cin >> n;
    a.resize(n);
    b.resize(n);
    for (auto& i : a) cin >> i;
    for (auto& i : b) cin >> i;
}

ll solve(int al, int ar, int bl, int br) {
    queue<int> qa, qb;
    for (int i = 0; i < n; i++) {
        if (al <= a[i] && a[i] <= ar) qa.push(i);
        if (bl <= b[i] && b[i] <= br) qb.push(i);
    }

    ll ret = 0;
    while (!qa.empty()) {
        ret += abs(qa.front() - qb.front());
        qa.pop();
        qb.pop();
    }
    return ret;
}

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    input();

    if (n & 1) {
        int mid = (n + 1) / 2;
        cout << min(solve(1, mid - 1, mid + 1, n), solve(1, mid, mid, n));
    }
    else cout << solve(1, n / 2, n / 2 + 1, n);
}
