#include <bits/stdc++.h>
using namespace std;

int n, m, d, st;
string s;

struct Edge {
    int v, p, l;
};
vector<Edge> adj[606];

void input() {
    cin >> n >> m >> d >> st;
    cin >> s;
    for (int i = 0; i < m; i++) {
        int u, v, p, l;
        cin >> u >> v >> p >> l;
        p--;
        adj[u].push_back({ v, p, l });
    }
}
