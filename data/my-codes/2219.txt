#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    int tc; cin >> tc;
    while (tc--) {
        int n; cin >> n;
        vector<ll> a(n);
        for (auto& i : a) cin >> i;

        int ans = 0;
        ll sum = 0;
        for (auto& i : a) {
            if (i == 0) continue;
            if (i > 0) {
                if (sum <= 0 && sum + i > 0) sum += i;
                else {
                    if (sum) ans += sum / abs(sum);
                    sum = i;
                }
            }
            if (i < 0) {
                if (sum > 0 && sum + i <= 0) {
                    if (sum) ans += sum / abs(sum);
                    sum = i;
                }
                else sum += i;
            }
        }
        if (sum) ans += sum / abs(sum);

        cout << (ans > 0 ? "YES" : "NO") << '\n';
    }
}
