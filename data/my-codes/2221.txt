int n;
vector<int> a;
ll dp[202020];

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    cin >> n;
    a.resize(n);
    for (auto& i : a) cin >> i;

    for (auto& i : a) dp[i]++;

    for (int i = 1; i < 101010; i++) {
        for (int j = 2; i * j < 101010; j++) {
            dp[i * j] += dp[i];
        }
    }

    int q; cin >> q;
    while (q--) {
        int x; cin >> x;
        cout << dp[x] << ' ';
    }
}
