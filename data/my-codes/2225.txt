const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

int n, m, q;
vector<ll> ro, co;

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    cin >> n >> m >> q;

    ro.resize(n);
    co.resize(m);

    while (q--) {
        int op, x, y;
        cin >> op >> x >> y;
        if (op == 1) ro[x - 1] += y;
        if (op == 2) co[x - 1] += y;
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << ro[i] + co[j] << ' ';
        }
        cout << '\n';
    }
}
