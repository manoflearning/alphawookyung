int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    int n; cin >> n;

    ll ans = 0;

    ans += (2 * n + 1) * (2 * n + 1);

    for (ll a = -n; a <= n; a++) {
        if (a == 0) continue;
        for (ll b = -n; b <= n; b++) {
            ll c = 1 - a - b;
            if (-n <= c && c <= n) ans++;
        }
    }

    cout << ans;
}
