const int MAXN = 1010;

int N, K;
vector<int> ans;
int sum[MAXN][MAXN];

int query(int L, int R) {
    assert(L <= R);

    cout << "? " << L << ' ' << R << '\n';
    cout << flush;

    int ret; cin >> ret;
    return ret;
}

void preprocessing() {
    for (int i = 1; i <= N; i++) {
        sum[1][i] = query(1, i);
    }
    for (int i = 1; i <= N; i++) {
        sum[i][N] = query(i, N);
    }
}

void solve() {
    for (int x = 1; x <= N; x++) {
        bool ls = false, rs = false;
        if (x == 1 || (x > 1 && sum[1][x - 1] < sum[1][x])) ls = true;
        if (x == N || (x < N && sum[x][N] > sum[x + 1][N])) rs = true;
        if (ls && rs) ans.push_back(x);
    }
}

int main() {

    cin >> N;

    preprocessing();

    solve();

    cout << "! " << sz(ans) << ' ';
    for (auto& i : ans) cout << i << ' ';
    cout << flush;
}
