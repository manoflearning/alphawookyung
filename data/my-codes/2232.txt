
#include <bits/stdc++.h>

const double EPS = 1e-9;
const int INF = 1e9 + 7;
const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

int N, K;

ll power(ll x, ll y) {
    if (y == 0) return 1;
    if (y == 1) return x;
    ll res = power(x, y / 2);
    return res * res % MOD * (y & 1 ? x : 1) % MOD;
}

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    cin >> N >> K;

    if (K == 0) {
        cout << N;
        return 0;
    }

    ll v = N, deg = N - 1;
    for (int i = 0; i < K; i++) {
        v = v * deg % MOD * power(2, MOD - 2) % MOD;
        deg = 2 * (deg - 1) % MOD;
    }

    cout << v;
}
