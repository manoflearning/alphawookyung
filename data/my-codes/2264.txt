int vi[10];

int bt(int idx, ll sum, const vt<ll>& b) {
    if (idx == 3) {
        int bit = 0;
        EACH(i, b) if (i == sum) bit = 1;
        return bit;
    }

    int ret = 1;
    FOR(sz(b)) {
        if (vi[i]) continue;
        vi[i] = 1;
        ret &= bt(idx + 1, sum + b[i], b);
        vi[i] = 0;
    }

    return ret;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int tc; cin >> tc;
    while (tc--) {
        int n; cin >> n;
        vt<ll> a(n);
        EACH(i, a) cin >> i;

        int pCnt = 0, nCnt = 0, cnt0 = 0;
        vt<ll> b;
        EACH(i, a) {
            if (i > 0) pCnt++;
            if (i < 0) nCnt++;

            if (i != 0) b.push_back(i);
            else cnt0++;
        }

        if (pCnt >= 3 || nCnt >= 3) {
            cout << "NO\n";
            continue;
        }

        FOR(min(3, cnt0)) b.push_back(0);
        
        if (bt(0, 0, b)) cout << "YES\n";
        else cout << "NO\n";
    }

	return 0;
}
