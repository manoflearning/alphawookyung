#define F_OR1(e) F_OR(i, 0, e, 1)
#define F_OR2(i, e) F_OR(i, 0, e, 1)
#define F_OR3(i, b, e) F_OR(i, b, e, 1)

#define F_ORC(...) EXPAND( GET5(__VA_ARGS__, F_OR4, F_OR3, F_OR2, F_OR1) )

const double EPS = 1e-9;

const int MOD = 1e9 + 7;

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int tc; cin >> tc;
	FOR(tt, 1, tc + 1) {
        ll n, m, sy, sx, d;
        cin >> n >> m >> sy >> sx >> d;

        if (sy - d <= 1 && n <= sy + d) {
            cout << -1 << '\n';
            continue;
        }
        if (sx - d <= 1 && m <= sx + d) {
            cout << -1 << '\n';
            continue;
        }

        if (sy - d <= 1 && sx - d <= 1) {
            cout << -1 << '\n';
            continue;
        }
        if (n <= sy + d && m <= sx + d) {
            cout << -1 << '\n';
            continue;
        }

        if (n - sy + m - sx <= d) {
            cout << -1 << '\n';
            continue;
        }

        cout << n + m - 2 << '\n';
	}

	return 0;
}
