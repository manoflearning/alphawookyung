void getbit(int& bit, const vector<int>& a, int x) {
    for (auto& i : a) {
        int nbit = bit | i;
        if (nbit - (nbit & x)) break;
        bit = nbit;
    }
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int tc; cin >> tc;
	for (int tt = 1; tt <= tc; tt++) {
        int n, x;
        cin >> n >> x;
        vector<int> a(n), b(n), c(n);
        for (auto& i : a) cin >> i;
        for (auto& i : b) cin >> i;
        for (auto& i : c) cin >> i;

        int b1 = 0, b2 = 0, b3 = 0;
        getbit(b1, a, x);
        getbit(b2, b, x);
        getbit(b3, c, x);

        if ((b1 | b2 | b3) == x) cout << "Yes\n";
        else cout << "No\n";
	}
}
