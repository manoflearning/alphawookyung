const int MAX = 202020;

bool isPrime[MAX + 1];
vector<int> prime(1, 2);

void getPrime() {
	fill(isPrime + 2, isPrime + MAX + 1, 1);

	for (ll i = 4; i <= MAX; i += 2)
		isPrime[i] = 0;
	for (ll i = 3; i <= MAX; i++) {
		if (isPrime[i]) prime.push_back(i);
		for (ll j = i * i; j <= MAX; j += i * 2)
			isPrime[j] = 0;
	}
}

int main() {

        cout << 2 << '\n';
        vt<int> vi(n + 1);
        FOR(i, 1, n + 1) {
            if (vi[i]) continue;
            ll x = i;
            while (x <= n) {
                cout << x << ' ';
                vi[x] = 1;
                x <<= 1;
            }
        }
        cout << '\n';
    }

	return 0;
}
