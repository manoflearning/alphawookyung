int RBS(const string& s) {
    int stk = 0;
    EACH(i, s) {
        if (i == '(') stk++;
        else {
            if (stk > 0) stk--;
            else return 0;
        }
    }
    return stk == 0;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int tc; cin >> tc;
    FOR(tt, 1, tc + 1) {
        string s; cin >> s;
    
        int cntq = 0;
        EACH(i, s) if (i == '?') cntq++;

        if (cntq <= 1) {
            cout << "YES\n";
            continue;
        }

        int n = sz(s);

        int cntl = 0, cntr = 0;
        EACH(i, s) {
            if (i == '(') cntl++;
            else if (i == ')') cntr++;
        }

        if (cntl == n / 2 || cntr == n / 2) {
            cout << "YES\n";
            continue;
        }

        int bit = 1;
        EACH(i, s) {
            if (i == '?') {
                if (cntl < n / 2 - 1) i = '(', cntl++;
                else if (cntl == n / 2 - 1 && bit) i = ')', cntr++, bit = 0;
                else if (cntl == n / 2 - 1 && !bit) i = '(', cntl++;
                else i = ')', cntr++;
            }
        }

        if (!RBS(s)) cout << "YES\n";
        else cout << "NO\n";
    }

	return 0;
}
