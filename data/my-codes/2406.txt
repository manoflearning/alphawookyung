int flag;  
struct Seg {  
	vector<ll> t;

	void build(int n) {
		for (flag = 1; flag < n; flag <<= 1);
		t.resize(2 * flag);

	}
	void modify(int p, ll value) {  
		for (t[p += flag - 1] = value; p > 1; p >>= 1) t[p >> 1] = max(t[p], t[p ^ 1]);
	}
	ll query(int l, int r, int n = 1, int nl = 1, int nr = flag) {  
		if (r < nl || nr < l) return 0;
		if (l <= nl && nr <= r) return t[n];

		int mid = (nl + nr) / 2;
		return max(query(l, r, n << 1, nl, mid), query(l, r, n << 1 | 1, mid + 1, nr));
	}
}seg;

ll n, m;
ll a[202020];

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n >> m;
    FOR(i, 1, m + 1) cin >> a[i];

    seg.build(m);
    FOR(i, 1, m + 1) seg.modify(i, a[i]);

    int q; cin >> q;
    while (q--) {
        ll sx, sy, ex, ey, k;
        cin >> sx >> sy >> ex >> ey >> k;
        
        if (abs(sx - ex) % k || abs(sy - ey) % k) {
            cout << "NO\n";
            continue;
        }
        if (a[sy] >= sx || a[ey] >= ex) {
            cout << "NO\n";
            continue;
        }

        ll h = seg.query(min(sy, ey), max(sy, ey));

        ll l = 0, r = n;
        while (l < r) {
            ll mid = (l + r) >> 1;
            if (k * mid + sx > h) r = mid;
            else l = mid + 1;
        }

        if (k * l + sx > n) {
            cout << "NO\n";
            continue;
        }

        cout << "YES\n";
    }

	return 0;
}
