ll binom(ll n, ll r) {
    r = min(r, n - r);

    ll ret = 1;
    for (ll i = 1; i <= r; i++) {
        ret *= n + 1 - i;
        ret /= i;
    }
    return ret;
}

int n, deg[5050], cnt[5050];

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n;
    for (int v = 1; v <= n; v++) {
        int cnt; cin >> cnt;
        while (cnt--) {
            int v; cin >> v;
            deg[v]++;
        }
    }

    ll sumA = 0, sumB = 0;
    for (int v = 1; v <= n; v++) {
        cnt[deg[v]]++;
        sumA += deg[v];
    }

    ll ans = 0;
    ll A = n, B = 0;
    for (int i = n - 1; i >= 0; i--) {
        int res = 0;
        for (int j = 0; j < cnt[i]; j++) {
            A--, B++;
            sumA -= i, sumB += i;
            if (sumA + B * (B - 1) == sumB && A && B) {
                ans += binom(cnt[i], j + 1);
            }
        }
    }

    cout << ans;

	return 0;
}
