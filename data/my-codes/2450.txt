        int n; cin >> n;

        FOR(n) {
			int x; cin >> x;

			auto it = a.find(x);
			if (it == a.end()) a.insert({ x, 1 });
			else it->sc++;

			while (x >= 1) {

				if (it != a.end()) {
					it->sc--;

					break;
				}
				x >>= 1;
			}
		}
