const double EPS = 1e-9;
const int INF = 1e9 + 7;
const int MOD = 998244353;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

int n, k, l;
ll ans[202020], a[202020], b[202020];

void push() {
    for (int i = 1; i <= n; i++) {
        a[i] = b[i];
        ans[i] = (ans[i] + b[i]) % MOD;
        b[i] = 0;
    }
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n >> k;
    for (int i = k; i <= n; i += k) b[i] = 1;
    l = k; push();

    for (int i = k + 1; ; i++) {
        if (l + i > n) break;
        for (int j = l + i; j <= n; j++)
            b[j] = (b[j - i] + a[j - i]) % MOD;
        l += i; push();
    }

    for (int i = 1; i <= n; i++)
        cout << ans[i] << ' ';

	return 0;
}
