#include <bits/stdc++.h>
using namespace std;

const int MAXD = 17;

int main() {
	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);
	
	int K, MAX = 1; 
	cin >> K;

	for (int i = 0; i < MAXD; i++) {
		if ((1 << i) & K) {
			MAX = (1 << (i + 1));
		}
	}

	cout << "3 3\n";
	cout << K + MAX << ' ' << K << ' ' << 0 << '\n';
	cout << MAX << ' ' << K << ' ' << 0 << '\n';
	cout << MAX << ' ' << K + MAX << ' ' << K << '\n';

	return 0;
}
