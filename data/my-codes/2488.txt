int main() {

    ios_base::sync_with_stdio(false);

    int tc; cin >> tc;
    for (int tt = 1; tt <= tc; tt++) {
        string s;
        cin >> s;

        string ans1, ans2;
        ans1.push_back(s[0]);
        bool isChanged = false;
        for (int i = 1; i < sz(s); i++) {
            if (!isChanged && s[i] == '0') ans1.push_back(s[i]);
            else {
                isChanged = true;
                ans2.push_back(s[i]);
            }
        }

        if (!ans1.empty() && !ans2.empty() && stoll(ans1) < stoll(ans2)) cout << ans1 << ' ' << ans2 << '\n';
        else cout << -1 << '\n';
    }
}
