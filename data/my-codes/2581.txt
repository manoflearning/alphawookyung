int n;
vt<int> a, v(26);

int f(int x, int y) {
    int prv = -1;
    FOR(n) {
        if (a[i] != x && a[i] != y) continue;

        if (a[i] == x) {
            if (prv == x) return 0;
            prv = x;
        }
        if (a[i] == y) {
            if (prv == y) return 0;
            prv = y;
        }
    }
    return 1;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int tc; cin >> tc;

    while (tc--) {
        string s; cin >> s;
        
        n = sz(s);
        a.resize(n);
        FOR(n) a[i] = s[i] - 'a';

        FOR(n) v[a[i]] = 1;

        int res = 1;
        FOR(26) {
            if (!v[i]) continue;
            FOR(j, i + 1, 26) {
                if (!v[j]) continue;
                res &= f(i, j);
            }
        }

        if (res) cout << "YES\n";
        else cout << "NO\n";

        FOR(26) v[i] = 0;
    }

	return 0;
}
