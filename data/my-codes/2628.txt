
#include <bits/stdc++.h>

const double EPS = 1e-9;
const int INF = 1e9 + 7;
const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int tc; cin >> tc;
	for (int tt = 1; tt <= tc; tt++) {
		int n;
        string s;
        cin >> n >> s;

        int L = 0, R = 0;
        for (auto& c : s) {
            if (c == 'L') L = 1;
            if (c == 'R') R = 1;
        }

        if (L && R) {
            int already = 0;
            for (int i = 0; i + 1 < sz(s); i++) {
                if (s[i] == 'R' && s[i + 1] == 'L') {
                    already = 1;
                }
            }

            if (already) {
                cout << 0 << '\n';
                continue;
            }

            for (int i = 0; i + 1 < sz(s); i++) {
                if (s[i] == 'L' && s[i + 1] == 'R') {
                    cout << i + 1 << '\n';
                    break;
                }
            }
        }
        else cout << -1 << '\n';
	}

	return 0;
}
