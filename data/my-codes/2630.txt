int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int tc; cin >> tc;
	for (int tt = 1; tt <= tc; tt++) {
		int n; cin >> n;

        if (n & 1) {
            if (n == 3) {
                cout << "NO\n";
                continue;
            }

            cout << "YES\n";
            int m = n / 2;
            for (int i = 0; i < n; i++) {
                cout << (i & 1 ? m : -(m - 1)) << ' ';
            }
            cout << '\n';
        }
        else {
            cout << "YES\n";
            for (int i = 0; i < n; i++) {
                cout << (i & 1 ? 1 : -1) << ' ';
            }
            cout << '\n';
        }
	}

	return 0;
}
