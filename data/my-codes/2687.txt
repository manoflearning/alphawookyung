int isFirst(const string& s, int l, int r) {
    int cnt = 0;
    for (int i = l; i <= r; i++) {
        if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u') cnt++;
    }
    return cnt == 2;
}

int isSecond(const string& s, int l, int r) {
    int cnt = 0;
    for (int i = l; i <= r; i++) {
        if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u') cnt++;
    }
    return cnt == 1;
}
