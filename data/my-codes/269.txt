#define F_OR1(e) F_OR(i, 0, e, 1)

int n, en, len, ans;
vt<pii> a;

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n;
    a.resize(n);
    EACH(i, a) cin >> i.fr >> i.sc;
    
    cin >> en >> len;

    a.push_back({ en, 0 });
    sort(all(a));

    priority_queue<int> pq;
    EACH(i, a) {
        while (len < i.fr) {
            if (pq.empty()) {
                cout << -1;
                return 0;
            }
            ans++;
            len += pq.top();
            pq.pop();
        }
        pq.push(i.sc);
    }

    cout << ans;

	return 0;
}
