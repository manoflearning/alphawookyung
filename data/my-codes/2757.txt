int flag;  
struct Seg {  
	vector<ll> t;

	void build(int n) {
        t.clear();
		for (flag = 1; flag < n; flag <<= 1);
		t.resize(2 * flag);

	}
	void modify(int p, ll value) {  
		for (t[p += flag - 1] = value; p > 1; p >>= 1) t[p >> 1] = t[p] + t[p ^ 1];
	}
	ll query(int l, int r, int n = 1, int nl = 1, int nr = flag) {  
		if (r < nl || nr < l) return 0;
		if (l <= nl && nr <= r) return t[n];

		int mid = (nl + nr) / 2;
		return query(l, r, n << 1, nl, mid) + query(l, r, n << 1 | 1, mid + 1, nr);
	}
}segx[4], seg;

int n, m;
int a[1010101];

ll query(int l1, int r1, int l2, int r2) {
    int diff = seg.query(l1, r1) - seg.query(l2, r2);
    
    int s3 = segx[3].query(l1, r1), s2 = segx[2].query(l1, r1);
    int e1 = segx[1].query(l2, r2), e2 = segx[2].query(l2, r2);

    int cnt2 = min(s3, e1);
    cnt2 = min(cnt2, diff / 4);
    diff -= cnt2 * 4;
    s3 -= cnt2, e1 -= cnt2;

    int cnt1 = min(s2, e1) + min(s3, e2);
    cnt1 = min(cnt1, diff / 2);
    diff -= cnt1 * 2;

    if (diff > 0) return -1;
    else return cnt1 + cnt2;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int tc; cin >> tc;
	FOR(tt, 1, tc + 1) {
		cout << "Case #" << tt << ": ";

        cin >> n >> m;

        seg.build(n);
        segx[1].build(n);
        segx[2].build(n);
        segx[3].build(n);

        for (int i = 1; i <= n; i++) {
            cin >> a[i];
            seg.modify(i, a[i]);
            segx[a[i]].modify(i, 1);
        }

        ll ans = 0;
        while (m--) {
            int x, y, z;
            cin >> x >> y >> z;

            segx[a[x]].modify(x, 0);
            a[x] = y;
            segx[a[x]].modify(x, 1);
            
            seg.modify(x, y);
            
            if (seg.query(1, z) > seg.query(z + 1, n)) ans += query(1, z, z + 1, n);
            if (seg.query(1, z) < seg.query(z + 1, n)) ans += query(z + 1, n, 1, z);
        }

        cout << ans << '\n';
	}

	return 0;
}
