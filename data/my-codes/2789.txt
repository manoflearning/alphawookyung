int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int n; cin >> n;
    string s; cin >> s;

    if (n == 2) {
        if (s[0] == s[1]) cout << "Yes";
        else cout << "No";
        return 0;
    }

    int ans = 1;
    for (int i = n / 2; i < n; i++) {
        if (s[i] != s[n - i - 1]) {
            if (i < n - 1 && s[i] == 'B' && (s[i + 1] == 'A' || s[i + 1] == 'B')) continue;
            else if (s[i] == 'A' && (s[i - 1] == 'A' || s[i - 1] == 'B')) continue;
            else ans = 0;
        }
    }

    if (ans) cout << "Yes";
    else cout << "No";

	return 0;
}
