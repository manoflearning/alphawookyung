
#include <bits/stdc++.h>

#define sc second
#define all(c) (c).begin(), (c).end()
#define sz(x) (int)(x).size()

const double EPS = 1e-9;
const int INF = 1e9 + 7;
const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

int f(const string& lhs, const string& rhs) {
	for (int i = 0; i < min(sz(lhs), sz(rhs)); i++) {
		if (lhs[i] != rhs[i]) return i;
	}
	return min(sz(lhs), sz(rhs));
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int n; cin >> n;
	vector<pair<string, int>> a(n);
	for (int i = 0; i < n; i++) {
		cin >> a[i].fr;
		a[i].sc = i;
	}

	sort(all(a));

	vector<int> ans(n);
	for (int i = 0; i < n; i++) {
		auto& res = ans[a[i].sc];
		if (i > 0) res = max(res, f(a[i - 1].fr, a[i].fr));
		if (i + 1 < n) res = max(res, f(a[i].fr, a[i + 1].fr));
	}

	for (auto& i : ans)
		cout << i << '\n';

	return 0;
}
