#define F_OR1(e) F_OR(i, 0, e, 1)
#define F_OR2(i, e) F_OR(i, 0, e, 1)

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	ll a, b, c, d, e, f, x;
    cin >> a >> b >> c >> d >> e >> f >> x;

    ll t1 = 0, t2 = 0, res1 = 0, res2 = 0;
    for (int i = 0; i < x; i++) {
        if (t1 < a) {
            t1++;
            res1 += b;
        }
        else {
            t1 = (t1 + 1) % (a + c);
        }

        if (t2 < d) {
            t2++;
            res2 += e;
        }
        else {
            t2 = (t2 + 1) % (d + f);
        }
    }

    if (res1 > res2) cout << "Takahashi";
    if (res1 < res2) cout << "Aoki";
    if (res1 == res2) cout << "Draw";

	return 0;
}
