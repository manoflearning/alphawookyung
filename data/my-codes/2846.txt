
#include <bits/stdc++.h>

const double EPS = 1e-9;
const int INF = 1e9 + 7;
const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

int n, m;
ll a[20][20];
map<ll, int> mp;

ll bt(int y, int x) {
    if (mp[a[y][x]] > 0) return 0;
    if (y == n && x == m) return 1;

    mp[a[y][x]]++;
    ll ret = 0;
    if (y < n) ret += bt(y + 1, x);
    if (x < m) ret += bt(y, x + 1);
    mp[a[y][x]]--;
    return ret;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> a[i][j];
        }
    }

    cout << bt(1, 1);

	return 0;
}
