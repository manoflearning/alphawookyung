ll a, x, MOD;

ll powuv(ll u, ll v) {
    if (v == 0) return 1;
    if (v == 1) return u % MOD;
    ll res = powuv(a, v / 2) % MOD;
    return res * res % MOD * (v & 1 ? u : 1) % MOD;
}

ll f(ll A, ll X) {
    if (X == 1) return 1;
    ll res = f(A, X / 2);
    ll ret = (res + res * powuv(A, X / 2)) % MOD;
    if (X & 1) {
        ret += powuv(A, X - 1);
        ret %= MOD;
    }
    return ret;
}
