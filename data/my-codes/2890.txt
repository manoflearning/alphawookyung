const double EPS = 1e-9;

const int MOD = 998244353;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

ll n, m, k;
ll dp[55][2505];

void init() {
    FOR(55) {
        FOR(j, 2505) {
            dp[i][j] = -1;
        }
    }
}

ll f(int idx, int lft) {
    ll& ret = dp[idx][lft];
    if (ret != -1) return ret;
    if (idx == n + 1) return ret = 1;

    ret = 0;
    FOR(i, 1, min<ll>(m + 1, lft + 1)) ret += f(idx + 1, lft - i);

    return ret %= MOD;
}

int main() {

	ios_base::sync_with_stdio(false);

    init();

	cin >> n >> m >> k;

}
