const int MAXN = 1010101;
const int INF = 1e9 + 7;

int n;
string s;

ll naive() {
    ll ret = 0;
    for (int l = 0; l < n; l++) {
        for (int r = l; r < n; r++) {
            int isAns = 1;
            stack<char> stk;
            for (int i = l; i <= r; i++) {
                if (s[i] == '(' || s[i] == '{') stk.push(s[i]);
                else {
                    if (stk.empty()) { isAns = 0; break; }
                    else if (stk.top() == '(' && s[i] == '}') { isAns = 0; break; }
                    else if (stk.top() == '{' && s[i] == ')') { isAns = 0; break; }
                    stk.pop();
                }
            }
            if (sz(stk)) isAns = 0;
            ret += isAns;
        }
    }
    return ret;
}

int cnt[2 * MAXN];
int stk[MAXN], pos[MAXN];
int po = 0;
ll group3() {
    ll ret = 0;

    int sum = MAXN;
    cnt[sum]++;
    for (int i = 0; i < n; i++) {
        if (s[i] == '(' || s[i] == '{') {
            stk[po] = i;
            pos[po] = sum;
            po++;
            sum++;
        }
        else {
            ret += (ll)cnt[sum] * (cnt[sum] - 1) / 2;
            cnt[sum] = 0;
            sum--;
            if (po == 0 || abs(s[i] - s[stk[po - 1]]) > 2) {
                for (int i = 0; i < po; i++) {
                    ret += (ll)cnt[pos[i]] * (cnt[pos[i]] - 1) / 2;
                    cnt[pos[i]] = 0;
                }
                po = 0;
            }
            else po--;
        }
        cnt[sum]++;
    }
    
    for (int i = 0; i < po; i++) {
        ret += (ll)cnt[pos[i]] * (cnt[pos[i]] - 1) / 2;
        cnt[pos[i]] = 0;
    }
    po = 0;
    ret += (ll)cnt[sum] * (cnt[sum] - 1) / 2;
    cnt[sum] = 0;

    return ret;
}
