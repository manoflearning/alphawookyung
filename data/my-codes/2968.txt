
#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define all(c) (c).begin(), (c).end()
#define sz(x) (int)(x).size()

const int MAXN = 3030;

int n;
ll a[MAXN];
ll dp[MAXN][MAXN], dp1[MAXN][MAXN], dp2[MAXN][MAXN];

void init() {}

void input() {
    cin >> n;
    a[0] = 0;
    for (int i = 1; i <= n; i++) cin >> a[i];
    a[n + 1] = 0;
}

void f() { 
    for (int i = 1; i <= n; i++) {
        dp[i][i] = a[i];
        dp1[i][i] = a[i] + a[i + 1];
        dp2[i][i] = a[i] + a[i - 1];
        
        dp1[i][i - 1] = a[i];
        dp2[i + 1][i] = a[i];
    }
    
    for (int len = 2; len <= n; len++) {
        for (int l = 1; ; l++) {
            int r = l + len - 1;
            if (r > n) break;

            dp[l][r] = 0;
            for (int i = l; i <= r; i++) {
                dp[l][r] = max(dp[l][r], min(dp1[l][i - 1], dp2[i + 1][r]));
            }
            dp1[l][r] = dp[l][r] + a[r + 1];
            dp2[l][r] = dp[l][r] + a[l - 1];
        }
    }
}

int main() {

    cin.tie(NULL);
    ios_base::sync_with_stdio(false);

    int tc; cin >> tc;
    for (int tt = 1; tt <= tc; tt++) {
        cout << "Case #" << tt << '\n';

        init();

        input();

        f();
        
        cout << dp[1][n] << '\n';

        cout << flush;
    }
}
