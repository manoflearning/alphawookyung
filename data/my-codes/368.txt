#include <iostream>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;

int n;
char word[10][9];
int abc_count[26];

bool compare(int a, int b);

int main() {
	scanf("%d", &n);

	for (int i = 0; i < n; i++) {
		scanf("%s", word[i]);

		int len = strlen(word[i]);

		for (int l = 0; l < len; l++)
			abc_count[word[i][l] - 65] += pow(10, len - l - 1);
	}

	sort(abc_count, abc_count + 26, compare);

	int ans = 0;
	for (int i = 0; i < 9; i++)
		ans += abc_count[i] * (9 - i);
	printf("%d", ans);

	return 0;
}

bool compare(int a, int b) {
	return a > b;
}
