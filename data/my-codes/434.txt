#define F_OR1(e) F_OR(i, 0, e, 1)
#define F_OR2(i, e) F_OR(i, 0, e, 1)
#define F_OR3(i, b, e) F_OR(i, b, e, 1)
#define F_OR4(i, b, e, s) F_OR(i, b, e, s)
#define GET5(a, b, c, d, e, ...) e
#define F_ORC(...) EXPAND( GET5(__VA_ARGS__, F_OR4, F_OR3, F_OR2, F_OR1) )

	cin >> n;
	a.resize(n);
	EACH(i, a) cin >> i;

int f() {
	int ret = 0;

	FOR(sz(a)) {
		int b = 0;

		int k = sz(a) - 1;
		FOR(j, sz(a)) {
			if (i == j) continue;

			while (k > 0 && (i == k || j == k || a[j] + a[k] > a[i])) k--;

			if (i != k && j != k && a[j] + a[k] == a[i]) b = 1;
		}
		
		ret += b;
	}
	return ret;
}

void solve() {
	input();

	sort(all(a));

	int ans = f();

	cout << ans;
