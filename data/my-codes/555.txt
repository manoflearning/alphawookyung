const double EPS = 1e-9;

const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };

ll n, l, c;

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

    cin >> n >> l >> c;
	
    ll cnt = 1;
    while ((cnt + 1) * l + cnt <= c) cnt++;
    if (cnt % 13 == 0) cnt--;
    
    ll ans = (n + cnt - 1) / cnt;
    ll x = n % cnt, y = n / cnt;
    
    if (x > 0 && x % 13 == 0) {
        if (y == 0) ans++;
        else {
            ll bit = 1;
            for (int i = 1; x + i < cnt; i++) {
                if ((cnt - i) % 13 && (x + i) % 13) bit = 0;
            }
            if (bit) ans++;
        }
    }

    cout << ans;

	return 0;
}
