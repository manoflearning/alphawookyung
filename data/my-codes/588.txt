
#include <bits/stdc++.h>
using namespace std;

int n;
int r[303][303];
int diff[303], ans[303]; 

int main() {
    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    memset(r, -1, sizeof(r));

    cin >> n;
    for (int i = 1; i <= n; i++) {
        for (int j = i; j <= n; j++) {
            cin >> r[i][j];
        }
    }

    diff[2] = r[1][2];
    for (int e = 3; e <= n; e++) {
        if (r[e - 1][e] == 0) {
            diff[e] = 0;
            continue;
        }

        int s = e - 1;
        while (s >= 2 && diff[s] == 0) s--;
        if (s < 2) {
            diff[e] = r[e - 1][e];
            continue;
        }

        s--;
        if (r[s][e] == r[s][e - 1] + r[e - 1][e]) {
            if (diff[s + 1] < 0) diff[e] = -r[e - 1][e];
            else diff[e] = r[e - 1][e];
        }
        if (r[s][e] != r[s][e - 1] + r[e - 1][e]) {
            if (diff[s + 1] < 0) diff[e] = r[e - 1][e];
            else diff[e] = -r[e - 1][e];
        }
    }

    ans[1] = 0;
    for (int i = 2; i <= n; i++) {
        ans[i] = diff[i] + ans[i - 1];
    }

    for (int i = 1; i <= n; i++) {
        cout << ans[i] << ' ';
    }
}
