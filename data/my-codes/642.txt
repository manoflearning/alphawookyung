const double EPS = 1e-9;
const int INF = 1e9 + 7;
const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

int n;
ll cnt;
vt<int> adj[1010101], cycle;
bool vi[1010101], fi[1010101];
int ind[1010101], len[1010101];

void input() {
    cin >> n >> cnt;
    FOR(n) {
        int u, v = i + 1;
        cin >> u;
        if (u ^ v) {
            adj[u].push_back(v);
            ind[v]++;
        }
    }
}

void dfs(int v) {
	vi[v] = 1;
	for (int next : adj[v]) {
		if (!vi[next]) dfs(next);
		else if (!fi[next]) cycle.push_back(next);
	}
	fi[v] = 1;
}

ll f(int v, int prv) {
    ll ret = (prv == -1 ? cnt : cnt - 1);

    EACH(i, adj[v]) {
        ret = ret * f(i, v) % MOD;
    }
    return ret;
}

bool isCycle[1010101];
int stk[1010101], top = 0;

int dfs2(int v) {
    vi[v] = 1;

    int ret = 0;
    EACH(i, adj[v]) {
        if (vi[i]) ret = 1;
        else ret |= dfs2(i);
    }
    
    if (ret) {
        stk[top++] = v;
        isCycle[v] = 1;
    }

    return ret;
}

ll f2(int v) {
    top = 0;
    dfs2(v);

    ll ret = 1;
    FOR(i, top) {
        int v = stk[i];
        EACH(j, adj[v]) {
            if (isCycle[j]) continue;
            ret = ret * f(j, v) % MOD;
        }
    }
    
    ret = ret * len[top] % MOD;
    
    return ret;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);
    
    input();

    FOR(i, 1, n + 1)
        if (!vi[i]) dfs(i);

    memset(vi, 0, sizeof(vi));

    ll ans = 1;

    FOR(i, 1, n + 1) {
        if (ind[i] == 0) {
            ans = ans * f(i, -1) % MOD;
        }
    }

    len[2] = cnt * (cnt - 1) % MOD;
    len[3] = cnt * (cnt - 1) % MOD * max(0ll, cnt - 2) % MOD;
    
    FOR(i, 4, n + 1) {
        len[i] = (len[i - 1] * max(0ll, cnt - 2) + len[i - 2] * (cnt - 1)) % MOD;
    }

    EACH(i, cycle) {
        ans = ans * f2(i) % MOD;
    }

    cout << ans;

	return 0;
}
