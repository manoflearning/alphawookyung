#include <bits/stdc++.h>
using namespace std;

int main() {

    cin.tie(NULL); cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    int n; cin >> n;
    int psum = 0, mn = 0, mx = 0;
    for (int i = 1; i <= n; i++) {
        int x; cin >> x;
        psum = (x == 1 ? 1 : -1) + psum;
        mn = min(mn, psum);
        mx = max(mx, psum);
    }
    cout << mx - mn;
}
