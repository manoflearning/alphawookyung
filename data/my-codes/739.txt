const double EPS = 1e-9;

const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };

ll exp(ll val, ll cnt) {
    if (cnt == 1) return val;
    ll res = exp(val, cnt / 2);
    return res * res % MOD * (cnt & 1 ? val : 1) % MOD;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	int n; cin >> n;
    ll ans = 0;
    for (int i = 0; i < n; i++) {
        ll x, y;
        cin >> x >> y;
        ans = (ans + y * exp(x, MOD - 2)) % MOD;
    }
    cout << ans;

	return 0;
}
