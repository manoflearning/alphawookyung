const double EPS = 1e-9;
const int INF = 1e9 + 7;
const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

int n, a[20202];
vt<int> b;
int visited[20202];

void input() {
    cin >> n;
    FOR(i, 1, n + 1) {
        cin >> a[i];
    }
}

ll gcd(ll a, ll b) {
	if (b == 0) return a;
	else return gcd(b, a % b);
}

ll lcm(ll a, ll b) {
	return a * b / gcd(a, b);
}

int dfs(int v, int st) {
    visited[v] = 1;
    if (v == st) return 0;
    return dfs(a[v], st) + 1;
}

int f() {
    int ret = 1;
    FOR(i, 1, n + 1) {
        if (!visited[i]) {
            ret = lcm(ret, dfs(a[i], i) + 1);
        }
    }

    return ret;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

    input();

    cout << f();

	return 0;
}
