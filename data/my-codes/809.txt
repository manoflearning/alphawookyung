int main() {

	ios_base::sync_with_stdio(false);

	int n; cin >> n;

    vector<ll> psum(2020202);
    for (int i = 0; i < n; i++) {
        int s, e;
        ll x, y;
        cin >> s >> e >> x >> y;
        psum[s] += x * y;
        psum[s + e] -= x * y;
    }

    for (int i = 1; i < 2020202; i++) {
        psum[i] += psum[i - 1];
    }

    cout << fixed;
    cout.precision(2);

    ll sum; cin >> sum;
    for (int i = 0; i < 2020202; i++) {
        if (sum > psum[i]) sum -= psum[i];
        else {
            cout << i + (double)sum / psum[i];
            return 0;
        }
    }

    cout << "OVERFLOW";

	return 0;
}
