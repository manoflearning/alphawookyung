const int MAX = 1e7;

int n;

int isPrime(ll x) {
    for (ll i = 3; i * i <= x; i += 2) {
        if (x % i == 0) return 0;
    }
    return 1;
}

void f(ll x, int d) {
    if (d == n) {
        cout << x << '\n';
        return;
    }
    
    FOR(i, 1, 10, 2) {
        if (isPrime(x * 10 + i)) {
            f(x * 10 + i, d + 1);
        }
    }
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n;

    f(2, 1);
    f(3, 1);
    f(5, 1);
    f(7, 1);

	return 0;
}
