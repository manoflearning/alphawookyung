	int n; cin >> n;
	cout << "YES\n";
	vector<int> ans(n + 1), ans2;
	int now = 1;
	for (int i = n - 1, bit = 0; i >= 1; i--, bit = !bit) {
		ans[now] = i;
		ans2.push_back(now);
		if (!bit) now += i;
		else now -= i;
	}
	ans[now] = n;
	ans2.push_back(now);
	for (int i = 1; i <= n; i++) cout << ans[i] << ' ';
	cout << '\n';
	for (auto& i : ans2)
		cout << i << ' ';

	return 0;
