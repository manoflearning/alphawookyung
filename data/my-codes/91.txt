
#include <bits/stdc++.h>

#define ll long long

const double EPS = 1e-9;

const int MOD = 1e9 + 7;
const int dy[] = { 0, 0, 1, -1, 1, 1, -1, -1 };
const int dx[] = { 1, -1, 0, 0, 1, -1, 1, -1 };

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	string s; cin >> s;
	int uc = 0, pd = 0;
	for (auto& i : s) {
		if (i == 'U' || i == 'C') uc++;
		else pd++;
	}

	string ans;
	if (uc > (pd + 1) / 2) {
		ans.push_back('U');
	}
	if (pd > 0) {
		ans.push_back('D');
		ans.push_back('P');
	}
	if (ans.empty())
		ans.push_back('C');
	cout << ans;

	return 0;
}
