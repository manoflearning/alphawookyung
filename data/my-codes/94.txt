int n, k;
vector<ll> a1, a2, a3;

ll f(vector<ll> x, vector<ll> y) {
    vector<ll> a(n);
    for (int i = 0; i < n; i++) {
        a[i] = x[i] + y[i];
    }

    sort(a.rbegin(), a.rend());
    ll ret = 0;
    for (int i = 0; i < k; i++) ret += a[i];
    return ret;
}

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n >> k;
    a1.resize(n);
    a2.resize(n);
    a3.resize(n);
    for (int i = 0; i < n; i++) 
        cin >> a1[i] >> a2[i] >> a3[i];

    ll ans = f(a1, a2);
    ans = max(ans, f(a1, a3));
    ans = max(ans, f(a2, a3));

    cout << ans;

	return 0;
}
