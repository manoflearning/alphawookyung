int n;
vt<int> cnt(101), cnt2(101);

int main() {

	cin.tie(NULL); cout.tie(NULL);
	ios_base::sync_with_stdio(false);

	cin >> n;
    FOR(i, 1, n + 1) {
        int x, y; 
        cin >> x >> y;
        cnt[x]++, cnt2[y]++;
        int sum = 0, sum2 = 0, r = 101, res = 0;
        FOR(l, 1, 101) {
            if (!cnt[l]) continue;
            sum += 1;
            while (1 < r && sum2 < sum) sum2 += cnt2[--r];
            sum += cnt[l] - 1;
            res = max(res, l + r);
        }
        cout << res << '\n';
    }

	return 0;
}
